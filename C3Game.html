What is the result of this recursive function?

#include 
using namespace std;

int sumList(int lst[], int size) {
    return (size == 0) ? 0 : lst[0] + sumList(lst + 1, size - 1);
}

int main() {
    int arr[] = {1, 2, 3, 4};
    cout << sumList(arr, 4);
    return 0;
}
