What is the result of this recursive function?

public class Main {
    public static int sumList(int[] lst, int size) {
        return (size == 0) ? 0 : lst[0] + sumList(lst, size - 1);
    }
    
    public static void main(String[] args) {
        int[] arr = {1, 2, 3, 4};
        System.out.println(sumList(arr, arr.length));
    }
}

What will this code output?

public class Circle {
    float radius;
    
    public Circle(float r) {
        radius = r;
    }
    
    public float area() {
        return 3.14f * radius * radius;
    }
    
    public static void main(String[] args) {
        Circle c = new Circle(5);
        System.out.println(c.area());
    }
}
